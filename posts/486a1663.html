Restore

发表于2023-05-20|更新于2023-06-09|DatabaseOracleRMAN
|阅读量:

Restore parameter file

rman target / nocatalog

RUN {
 ALLOCATE CHANNEL ch00 TYPE DISK;
 RESTORE SPFILE FROM '/BACKUP/SPFILE_01UUF10M_1_1.BAK' TO PFILE '/tmp/initTEMP.ora';
 RELEASE CHANNEL ch00;
}

STARTUP FORCE PFILE='/tmp/initTEMP.ora';

Restore control file

rman target / nocatalog

startup nomount;
set dbid 320066378;

run {
 allocate channel ch00 type disk;
 restore controlfile from '/backup/cntrl_46_1_730733685';
 release channel ch00;
}

Restore database file

rman target / nocatalog

sql 'alter database mount';
catalog start with '/backup/'; -- 将备份集信息重新导入到当前控制文件中 

run {
 allocate channel ch00 type disk;
 allocate channel ch01 type disk;
 -- specify the new location for each datafile
 set newname for database to '/datafile/%U.dbf';
 -- set newname for datafile 1 to '/oracle/oradata/datafile01.dbf';
 -- set newname for tempfile 1 to '/oracle/oradata/tempfile01.dbf';
 restore database;
 switch datafile all; -- update control file with new filenames
 switch tempfile all;
 -- recover database;
 release channel ch00;
 release channel ch01;
}

Restore tablespace

RUN
{
 SQL 'ALTER TABLESPACE users OFFLINE IMMEDIATE';
 SQL 'ALTER TABLESPACE tools OFFLINE IMMEDIATE';
 -- specify the new location for each datafile
 SET NEWNAME FOR DATAFILE '/olddisk/users01.dbf' TO '/newdisk/users01.dbf';
 SET NEWNAME FOR DATAFILE '/olddisk/tools01.dbf' TO '/newdisk/tools01.dbf';
 -- to restore to an ASM disk group named dgroup, use:
 -- SET NEWNAME FOR DATAFILE '/olddisk/trgt/tools01.dbf' TO '+dgroup';
 RESTORE TABLESPACE users, tools;
 SWITCH DATAFILE ALL; -- update control file with new filenames
 RECOVER TABLESPACE users, tools;
}

SQL 'ALTER TABLESPACE users ONLINE';
SQL 'ALTER TABLESPACE tools ONLINE';

Recover archivelog

rman target / nocatalog

run {
 allocate channel ch00 type disk;
 set archivelog destination to '/backup/arch/';
 -- restore archivelog from logseq 10 until logseq 14 thread 1;
 -- restore archivelog from logseq 10 until logseq 14 thread 2;
 recover database;
 release channel ch00;
}

Recover database using backup controlfile until time

sqlplus "/as sysdba"
alter system set log_archive_dest='/oracle/arch/';
recover database using backup controlfile until time 'JUN 29 2011 21:30:00';

Recover database using backup controlfile

Rename online redo

select 'alter database rename file''' || member || '''to''/dest_dir/' || substr(member, instr(member, '/', -1) + 1) || ''';' from v$logfile;

Rename datafile and tempfile

select 'alter database rename file''' || member || '''to''/dest_dir/' || substr(member, instr(member, '/', -1) + 1) || ''';' from v$datafile;
select 'alter database rename file''' || member || '''to''/dest_dir/' || substr(member, instr(member, '/', -1) + 1) || ''';' from v$tempfile;

Rename datafile and tempfile on rman script

select 'set newname for datafile''' || name|| '''to''/dest_dir/' || substr(name, instr(name, '/', -1) + 1) || ''';' from v$datafile order by file#;
select 'set newname for tempfile''' || name || '''to''/dest_dir/' || substr(name, instr(name, '/', -1) + 1) || ''';' from v$tempfile order by file#;

Open database with resetlogs

sqlplus "/as sysdba"
SQL> alter database open resetlogs;

Troubleshooting

open resetlogs 出现 system 表空间没有从一个完整的旧备份中还原错误，解决方法参考 onenote 笔记 ORA-01152
