xtarbackup

Full Backup

# 备份 
mkdir -p /xtarbackup/full
innobackupex --user=root --password='welcome@123' /xtarbackup/full

# 停止 MySQL 服务，删除数据 
systemctl stop mysqld
rm -rf /var/lib/mysql/*
rm -rf /var/log/mysqld.log
rm -rf /var/log/mysql-slow/slow.log

# 备份还原 
innobackupex --apply-log /xtarbackup/full/2022-01-01_00-00-00
innobackupex --copy-back /xtarbackup/full/2022-01-01_00-00-00 /var/lib/mysql
chown -R mysql.mysql /var/lib/mysql
systemctl start mysqld

Incremental Backup

# 增量备份 
innobackupex --user=root --password='welcome@123' \
--incremental /xtarbackup/incr/ \
--incremental-basedir=/xtarbackup/full/2022-01-01_00-00-00/
innobackupex --user=root --password='welcome@123' \
--incremental /xtarbackup/incr/ \
--incremental-basedir=/xtarbackup/full/2022-01-02_00-00-00/

# 停止 MySQL 服务，删除数据 
systemctl stop mysqld
rm -rf /var/lib/mysql/*
rm -rf /var/log/mysqld.log
rm -rf /var/log/mysql-slow/slow.log

# 备份还原 
innobackupex --apply-log --redo-only /xtarbackup/full/2022-01-01_00-00-00/
innobackupex --apply-log --redo-only /xtarbackup/full/2022-01-01_00-00-00/ \
--increment-dir=/xtarbackup/incr/2022-01-02_00-00-00
innobackupex --apply-log --redo-only /xtarbackup/full/2022-01-01_00-00-00/ \
--increment-dir=/xtarbackup/incr/2022-01-03_00-00-00
innobackupex --copy-back /xtarbackup/full/2022-01-01_00-00-00 /var/lib/mysql
chown -R mysql.mysql /var/lib/mysql
systemctl start mysqld

# 差异备份
