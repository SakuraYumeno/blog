Ansible 配置文件

发表于2023-05-20|更新于2023-06-09|LinuxAnsible
|阅读量:

Ansible 按照如下位置和顺序来查找 ansible.cfg 文件：

  1. ANSIBLE_CONFIG 环境变量所指定的文件
  2. 当前目录下的 ansible.cfg 文件
  3. Home 目录下的的.ansible.cfg 文件
  4. /etc/ansible/ansible.cfg 文件 
[defaults]
inventory = ./inventory
remote_user = normaluser
ask_pass = false

[privilege_escalation]
become = true
become_method = sudo
become_user = root
become_ask_pass = false

ask_pass = false 需要配置 ssh 免密登录；
become_ask_pass = false 需要在 sudoers 配置免密提权。
